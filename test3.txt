const readline = require('readline');
const inp = readline.createInterface({
  input: process.stdin
});
const userInput = [];
  inp.on("line", (data) => {
  userInput.push(data);
});
inp.on("close", () => {
var string=userInput[0];
  var freq=[];
  for(var i=0;i<string.lenght;i++)
  {
     var character=string.charAt(i);
    if(freq[character])
    {
      freq[character]++;
    }
    else
    {
      freq[character]=1;
    }
  }
  
  console.log(freq);
  
  
})
